    agent any
    stages {
        stage('Checkout SCM') {
            steps {
                checkout([$class: 'GitSCM',

                        extensions: [], 
                        userRemoteConfigs: [[url: 'https://github.com/MarkEWaite/git-client-plugin.git']]])
            }
        }
    }
}
